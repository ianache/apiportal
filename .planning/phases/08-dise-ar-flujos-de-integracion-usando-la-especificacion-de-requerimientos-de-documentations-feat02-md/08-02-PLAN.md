Following the `registry.ts` pattern exactly — `defineStore`, state with `loading`/`error`, actions using `useAuthStore().getToken()` and `fetch` to `bffBase`:

```typescript
import { defineStore } from 'pinia';
import { useAuthStore } from './auth';
import type { IntegrationFlow, SubFlowType, FlowNodeData } from 'shared-types';

interface FlowNode {
  id: string;
  type: string;
  position: { x: number; y: number };
  data: FlowNodeData;
}

interface FlowEdge {
  id: string;
  source: string;
  target: string;
  sourceHandle?: string;
  targetHandle?: string;
}

export const useFlowDesignerStore = defineStore('flowDesigner', {
  state: () => ({
    flows: [] as IntegrationFlow[],
    flow: null as IntegrationFlow | null,
    activeSubFlow: 'incoming' as SubFlowType,
    subFlows: {
      incoming: { nodes: [] as FlowNode[], edges: [] as FlowEdge[] },
      response: { nodes: [] as FlowNode[], edges: [] as FlowEdge[] },
      exception: { nodes: [] as FlowNode[], edges: [] as FlowEdge[] },
    },
    selectedNodeId: null as string | null,
    loading: false,
    error: null as string | null,
  }),
  getters: {
    activeNodes: (state) => state.subFlows[state.activeSubFlow].nodes,
    activeEdges: (state) => state.subFlows[state.activeSubFlow].edges,
    selectedNode: (state) => {
      if (!state.selectedNodeId) return null;
      return state.subFlows[state.activeSubFlow].nodes.find(n => n.id === state.selectedNodeId) ?? null;
    },
  },
  actions: {
    setActiveSubFlow(type: SubFlowType) { this.activeSubFlow = type; },
    updateNodes(nodes: FlowNode[]) { this.subFlows[this.activeSubFlow].nodes = nodes; },
    updateEdges(edges: FlowEdge[]) { this.subFlows[this.activeSubFlow].edges = edges; },
    toPayload() {
      return {
        incomingFlow: this.subFlows.incoming,
        responseFlow: this.subFlows.response,
        exceptionFlow: this.subFlows.exception,
      };
    },
    async fetchFlows() { /* GET /flows, set this.flows */ },
    async fetchFlow(id: string) { /* GET /flows/:id, set this.flow + populate subFlows */ },
    async createFlow(payload: { name: string; description?: string; tags?: string[] }) { /* POST /flows */ },
    async deleteFlow(id: string) { /* DELETE /flows/:id */ },
    async saveFlowConfig(id: string) { /* PUT /flows/:id/config with toPayload() */ },
  }
});
```

Each async action follows the registry.ts pattern: set loading=true, get token from auth store, fetch to `${bffBase}/flows`, handle errors, set loading=false in finally.

Per FLOW-01 and FLOW-02: Store manages both catalog listing and designer canvas state.
